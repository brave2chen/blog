```docker
# elasticsearch容器开机自动允许
docker container update --restart=always  elasticsearch

# 查看docker网络
docker network ls

# elasticsearch连接到docker-net的网络
docker network connect docker-net elasticsearch

# 查看容器IP地址，映射端口信息
docker inspect -f='{{.Name}} {{.NetworkSettings.IPAddress}} {{.HostConfig.PortBindings}}' $(docker ps -aq)
```
